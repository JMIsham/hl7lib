006package org.dellroad.hl7;
007
008import java.io.Serializable;
009import java.util.ArrayList;
010
011/**
012 * Represents one segment in an HL7 message.
013 */
014@SuppressWarnings("serial")
015public class HL7Segment implements Serializable {
016
017    protected final ArrayList<HL7Field> fields = new ArrayList<HL7Field>();
018
019    /**
020     * Constructor for {@link MSHSegment} use only.
021     */
022    HL7Segment() {
023    }
024
025    /**
026     * Constructor that initializes the name of the segment only.
027     *
028     * @param name segment name
029     * @throws HL7ContentException if the name is an invalid segment name
030     */
031    public HL7Segment(String name) throws HL7ContentException {
032        this.setName(name);
033    }
034
035    /**
036     * Constructor taking segment name and explicit fields.
037     *
038     * @param name segment name
039     * @param fields zero or more segment fields
040     * @throws HL7ContentException if the name is an invalid segment name
041     * @throws IllegalArgumentException if any field in the array is null
042     */
043    public HL7Segment(String name, HL7Field[] fields) throws HL7ContentException {
044        this(name);
045        for (HL7Field field : fields) {
046            if (field == null)
047                throw new IllegalArgumentException("null field");
048            this.fields.add(field);
049        }
050    }
051
052    /**
053     * Constructor that parses a segment line.
054     *
055     * @param line segment line
056     * @param seps separator and escape characters
057     * @throws HL7ContentException if the segment name is invalid
058     */
059    public HL7Segment(String line, HL7Seps seps) throws HL7ContentException {
060        this.parseAndAddFields(line, seps);
061    }
062
063    /**
064     * Copy constructor.
065     */
066    public HL7Segment(HL7Segment segment) {
067        for (HL7Field field : segment.getFields())
068            this.appendField(new HL7Field(field));
069    }
070
071    /**
072     * Get segment name, e.g., <code>MSH</code>, <code>PV1</code>, etc.
073     * Equivalent to:
074     * <blockquote>
075     * <code>getField(0).get(0, 0, 0)</code>
076     * </blockquote>
077     */
078    public String getName() {
079        return this.fields.get(0).get(0, 0, 0);
080    }
081
082    /**
083     * Set segment name.
084     *
085     * @param name new segment name
086     * @throws HL7ContentException if the name is an invalid segment name
087     */
088    public void setName(String name) throws HL7ContentException {
089        this.checkSegmentName(name);
090        HL7Field field = new HL7Field(name);
091        if (this.fields.isEmpty())
092            this.fields.add(field);
093        else
094            this.fields.set(0, field);
095    }
096
097    /**
098     * Get number of fields (including segment name).
099     */
100    public int getNumFields() {
101        return this.fields.size();
102    }
103
104    /**
105     * Add a field to the end of this segment.
106     *
107     * @param field field to add
108     * @throws IllegalArgumentException if field is null
109     */
110    public void appendField(HL7Field field) {
111        if (field == null)
112            throw new IllegalArgumentException("field is null");
113        this.fields.add(field);
114    }
115
116    /**
117     * Set a field in this segment. If the segment doesn't have <code>index</code> fields, intermediate empty fields will
118     * be added.
119     *
120     * <p>
121     * This method cannot be used to set the segment name. Use {@link #setName} for that.
122     *
123     * @param field field to set
124     * @param index index of the field in the segment (where zero is the segment name); must be at least one
125     * @throws IllegalArgumentException if field is null
126     * @throws IllegalArgumentException if index is zero or less
127     */
128    public void setField(int index, HL7Field field) {
129        if (field == null)
130            throw new IllegalArgumentException("field is null");
131        if (index < 1)
132            throw new IllegalArgumentException("index=" + index);
133        while (this.fields.size() < index + 1)
134            this.fields.add(HL7Field.EMPTY);
135        this.fields.set(index, field);
136    }
137
138    /**
139     * Convenience method. Equivalent to:
140     * <blockquote>
141     * <code>setField(index, new HL7Field(value))</code>
142     * </blockquote>
143     *
144     * @param value
145     */
146    public void setField(int index, String value) {
147        this.setField(index, new HL7Field(value));
148    }
149
150    /**
151     * Get a field by "HL7 index". E.g., if this message is "ABC", an index of 3 would return ABC.3. If index is zero, the name
152     * of this segment is returned.
153     *
154     * @param index of the requested field
155     * @return the requested field, or null if not present in this segment
156     * @throws IllegalArgumentException if <code>index</code> is negative
157     */
158    public HL7Field getField(int index) {
159        if (index < 0)
160            throw new IllegalArgumentException("index=" + index);
161        if (index >= this.fields.size())
162            return null;
163        return this.fields.get(index);
164    }
165
166    /**
167     * Get all of the fields, starting with the segment name.
168     */
169    public HL7Field[] getFields() {
170        return this.fields.toArray(new HL7Field[this.fields.size()]);
171    }
172
173    /**
174     * Trim off all but the first <code>size</code> fields. If the current number of fields is already
175     * less than or equal to <code>size</code> then nothing happens.
176     *
177     * @throws IllegalArgumentException if <code>size</code> is less than one
178     */
179    public void trimTo(int size) {
180        if (size < 1)
181            throw new IllegalArgumentException("size < 1");
182        while (size < this.fields.size())
183            this.fields.remove(this.fields.size() - 1);
184        this.fields.trimToSize();
185    }
186
187    /**
188     * Append string encoding of this segment to the provided buffer.
189     *
190     * @param seps HL7 separator and escape characters
191     */
192    public void append(StringBuilder buf, HL7Seps seps) {
193        boolean first = true;
194        for (HL7Field field : this.fields) {
195            if (first)
196                first = false;
197            else
198                buf.append(seps.getFieldSep());
199            field.append(buf, seps);
200        }
201    }
202
203    /**
204     * Convert to a string using the provided separators.
205     *
206     * @param seps HL7 separator and escape characters
207     */
208    public String toString(HL7Seps seps) {
209        StringBuilder buf = new StringBuilder();
210        this.append(buf, seps);
211        return buf.toString();
212    }
213
214    /**
215     * Convert to a string using {@link HL7Seps#DEFAULT}.
216     */
217    @Override
218    public String toString() {
219        return this.toString(HL7Seps.DEFAULT);
220    }
221
222    @Override
223    public boolean equals(Object obj) {
224        if (obj == null || obj.getClass() != getClass())
225            return false;
226        HL7Segment that = (HL7Segment)obj;
227        return this.fields.equals(that.fields);
228    }
229
230    @Override
231    public int hashCode() {
232        return this.fields.hashCode();
233    }
234
235    /**
236     * Parse fields from given line and add them.
237     *
238     * @param line segment or segment fragment
239     * @param seps separator and escape characters
240     * @throws HL7ContentException if this segment contains zero fields and the first field is not a valid segment name
241     */
242    protected void parseAndAddFields(String line, HL7Seps seps) throws HL7ContentException {
243        int[] fieldSeps = HL7Util.find(line, seps.getFieldSep());
244        int posn = 0;
245        for (int i = 0; i < fieldSeps.length; i++) {
246            String string = line.substring(posn, fieldSeps[i]);
247            if (this.fields.isEmpty())
248                this.checkSegmentName(string);
249            this.fields.add(new HL7Field(string, seps));
250            posn = fieldSeps[i] + 1;
251        }
252    }
253
254    /**
255     * Verify segment name is legit.
256     */
257    private void checkSegmentName(String name) throws HL7ContentException {
258        if (name.length() != 3)
259            throw new HL7ContentException("invalid segment name `" + name + "'");
260    }
261}
